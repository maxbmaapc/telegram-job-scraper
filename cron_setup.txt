# Telegram Job Scraper Cron Job Setup
# This runs the scraper every 30 minutes

# Edit your crontab:
# crontab -e

# Add this line to run every 30 minutes:
# */30 * * * * cd /Users/maxbmaapc/GitHub/telegram-job-scraper && /Users/maxbmaapc/GitHub/telegram-job-scraper/venv/bin/python run.py --mode single >> /Users/maxbmaapc/GitHub/telegram-job-scraper/cron.log 2>&1

# Or run every hour:
# 0 * * * * cd /Users/maxbmaapc/GitHub/telegram-job-scraper && /Users/maxbmaapc/GitHub/telegram-job-scraper/venv/bin/python run.py --mode single >> /Users/maxbmaapc/GitHub/telegram-job-scraper/cron.log 2>&1

# Or run every 2 hours:
# 0 */2 * * * cd /Users/maxbmaapc/GitHub/telegram-job-scraper && /Users/maxbmaapc/GitHub/telegram-job-scraper/venv/bin/python run.py --mode single >> /Users/maxbmaapc/GitHub/telegram-job-scraper/cron.log 2>&1

# To view cron logs:
# tail -f /Users/maxbmaapc/GitHub/telegram-job-scraper/cron.log

# To list current cron jobs:
# crontab -l

# To remove all cron jobs:
# crontab -r
